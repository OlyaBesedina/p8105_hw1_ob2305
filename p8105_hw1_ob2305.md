my_df = tibble(
  vec_numeric = rnorm(8),
  vec_logical = vec_numeric > 0,
  vec_char = c("monkey", "eel", "finch", "dog", "cat", "salmon", "carp","parrot"),
  vec_factor = factor(c("mammal", "fish", "bird", "mammal", "mammal", "fish", "fish", "bird"))
)

# The mean of the variables inside the data frame
